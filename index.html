$(document).ready(function(){ $("#search").keyup(function(){ let query = $(this).val(); if(query.length > 1){ $.ajax({ url: "search.php", method: "GET", data: {q: query}, success: function(data){ $("#suggestions").html(data); $("#suggestions").show(); } }); } else { $("#suggestions").hide(); } }); // Click suggestion $(document).on("click", "#suggestions div", function(){ $("#search").val($(this).text()); $("#suggestions").hide(); }); });
